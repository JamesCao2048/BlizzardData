import java.io.IOException;
// example 3

(com.foo.framework.persistence.*);

// try ... catch block
after() throwing (IOException e) : within(test.*) {
throw new PersistenceException(e);
}
declare soft: (IOException): module();
}
// tested by: test.ConvertToUnchecked
